-- ========== HELPER FUNCTIONS ==========
local function clamp(v,a,b)
    return math.max(a, math.min(b, v))
end

local function round(num, decimal)
    local mult = 10^(decimal or 1)
    return math.floor(num * mult + 0.5) / mult
end

local function timeFmt(s)
    s = math.floor(s or 0)
    return string.format("%02d:%02d", math.floor(s/60), s%60)
end

local function timeFmtDetailed(s)
    local minutes = math.floor(s / 60)
    local seconds = math.floor(s % 60)
    local centiseconds = math.floor((s % 1) * 100)
    return string.format("%02d:%02d.%02d", minutes, seconds, centiseconds)
end

local function safeClipboard(txt)
    local success = false
    if pcall(function() setclipboard(txt) end) then
        success = true
    elseif pcall(function() clipboard.set(txt) end) then
        success = true
    elseif pcall(function() writeclipboard(txt) end) then
        success = true
    end
    return success
end

local function readFromClipboard()
    local data = nil
    if pcall(function() data = readclipboard() end) then
        return data
    elseif pcall(function() data = clipboard.get() end) then
        return data
    elseif pcall(function() data = getclipboard() end) then
        return data
    end
    return nil
end

-- ========== HTTP REQUEST FUNCTION ==========
local function httpGet(url)
    local httpFunc = syn and syn.request or request or http_request
    if not httpFunc then
        return nil, "HTTP request not supported"
    end
    
    local success, response = pcall(function()
        return httpFunc({
            Url = url,
            Method = "GET",
            Headers = {
                ["Content-Type"] = "application/json"
            }
        })
    end)
    
    if not success then
        return nil, "Request failed"
    end
    
    if response and (response.Body or response.Data) then
        return response.Body or response.Data, nil
    end
    
    return nil, "No data received"
end

-- ========== PATH RECORD FUNCTIONS ==========
local function clearPathRecord()
    for _, part in ipairs(pathRecordParts) do
        pcall(function() part:Destroy() end)
    end
    for _, beam in ipairs(pathRecordBeams) do
        pcall(function() beam:Destroy() end)
    end
    pathRecordParts = {}
    pathRecordBeams = {}
end

local function createPathPart(position, color)
    local part = Instance.new("Part")
    part.Size = Vector3.new(0.4, 0.4, 0.4)
    part.Position = position
    part.Anchored = true
    part.CanCollide = false
    part.CanQuery = false
    part.CanTouch = false
    part.Material = Enum.Material.Neon
    part.Color = color
    part.Transparency = 0.3
    part.Parent = Workspace
    return part
end

local function createBeamBetween(startPos, endPos, color)
    local part = Instance.new("Part")
    part.Anchored = true
    part.CanCollide = false
    part.CanQuery = false
    part.CanTouch = false
    part.Transparency = 1

    part.Color = color
    part.Material = Enum.Material.Neon
    part.Transparency = 0.5
    part.Parent = Workspace
    
    return part
end

local function showPathRecord()
    if not pathRecordActive then return end
    if #recordedFrames < 2 then
        showToast("⚠️ No recording to show path!")
        return
    end
    
    clearPathRecord()
    
    local colors = {
        Color3.fromRGB(255, 80, 80),
        Color3.fromRGB(255, 180, 80),
        Color3.fromRGB(80, 255, 80),
        Color3.fromRGB(80, 200, 255),
        Color3.fromRGB(180, 80, 255)
    }
    
    local totalFrames = #recordedFrames
    local step = math.max(1, math.floor(totalFrames / 200))
    
    for i = 1, totalFrames, step do
        local frame = recordedFrames[i]
        if frame and frame.pos then
            local progress = i / totalFrames
            local colorIndex = math.floor(progress * (#colors - 1)) + 1
            local color = colors[math.min(colorIndex, #colors)]
            local part = createPathPart(frame.pos, color)
            table.insert(pathRecordParts, part)
        end
    end
    
    for i = 1, totalFrames - 1 do
        local frame1 = recordedFrames[i]
        local frame2 = recordedFrames[i + 1]
        
        if frame1 and frame2 and frame1.pos and frame2.pos then
            local progress = i / totalFrames
            local colorIndex = math.floor(progress * (#colors - 1)) + 1
            local color = colors[math.min(colorIndex, #colors)]
            local beamPart = createBeamBetween(frame1.pos, frame2.pos, color)
            table.insert(pathRecordBeams, beamPart)
        end
    end
    
    if recordedFrames[1] and recordedFrames[1].pos then
        local startMarker = Instance.new("Part")
        startMarker.Size = Vector3.new(1, 1, 1)
        startMarker.Position = recordedFrames[1].pos
        startMarker.Anchored = true
        startMarker.CanCollide = false
        startMarker.Color = Color3.fromRGB(0, 255, 0)
        startMarker.Material = Enum.Material.Neon
        startMarker.Parent = Workspace
        table.insert(pathRecordParts, startMarker)
    end
    
    if recordedFrames[totalFrames] and recordedFrames[totalFrames].pos then
        local endMarker = Instance.new("Part")
        endMarker.Size = Vector3.new(1, 1, 1)
        endMarker.Position = recordedFrames[totalFrames].pos
        endMarker.Anchored = true
        endMarker.CanCollide = false
        endMarker.Color = Color3.fromRGB(255, 0, 0)
        endMarker.Material = Enum.Material.Neon
        endMarker.Parent = Workspace
        table.insert(pathRecordParts, endMarker)
    end
    
    showToast("✨ Path created with " .. #recordedFrames .. " frames!")
end

local function togglePathRecord()
    if pathRecordActive then
        clearPathRecord()
        pathRecordActive = false
        showToast("📉 Path record: OFF")
    else
        if #recordedFrames < 2 then
            showToast("⚠️ No recording to show path! Record something first.")
            return
        end
        pathRecordActive = true
        showPathRecord()
        showToast("📈 Path record: ON")
    end
end

-- ========== SMART PLAYBACK RESUME FUNCTIONS ==========
local function findClosestFrameToPosition(position)
    if #recordedFrames == 0 then return 1, 0 end
    
    local closestIndex = 1
    local closestDistance = (recordedFrames[1].pos - position).Magnitude
    
    for i = 2, #recordedFrames do
        local distance = (recordedFrames[i].pos - position).Magnitude
        if distance < closestDistance then
            closestDistance = distance
            closestIndex = i
        end
    end
    
    return closestIndex, closestDistance
end

local function findFrameAtTime(time)
    if #recordedFrames == 0 then return 1 end
    
    for i = 1, #recordedFrames - 1 do
        if time >= recordedFrames[i].t and time <= recordedFrames[i+1].t then
            return i
        end
    end
    
    return #recordedFrames
end

local function updatePlaybackSpeed(newWalkSpeed)
    currentPlaybackWalkSpeed = clamp(newWalkSpeed, MIN_PLAYBACK_SPEED, MAX_PLAYBACK_SPEED)
    currentPlaybackWalkSpeed = round(currentPlaybackWalkSpeed, 1)
    speedBox.Text = string.format("%.1f", currentPlaybackWalkSpeed)
    
    local sliderSize = speedSliderFrame.AbsoluteSize.X
    if sliderSize > 0 then
        local t = (currentPlaybackWalkSpeed - MIN_PLAYBACK_SPEED) / (MAX_PLAYBACK_SPEED - MIN_PLAYBACK_SPEED)
        local relativeX = t * sliderSize
        speedSliderKnob.Position = UDim2.new(relativeX / sliderSize, -6, 0.5, -6)
    end
end

    if speedSliderDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local mousePos = input.Position
        local sliderPos = speedSliderFrame.AbsolutePosition
        local sliderSize = speedSliderFrame.AbsoluteSize.X
        
        if sliderSize > 0 then
            local relativeX = math.clamp(mousePos.X - sliderPos.X, 0, sliderSize)
            local t = relativeX / sliderSize
            local speedValue = MIN_PLAYBACK_SPEED + t * (MAX_PLAYBACK_SPEED - MIN_PLAYBACK_SPEED)
            speedValue = round(speedValue, 1)
            updatePlaybackSpeed(speedValue)
        end
    end
end)

        v = round(v, 1)

    updatePlaybackSpeed(v)

    end
end)

-- ========== FUNGSI GET CHARACTER DAN GET VELOCITY ==========
local function getChar()
    local c = player.Character
    if not c then return nil,nil,nil end
    local hum = c:FindFirstChildOfClass("Humanoid")
    local hrp = c:FindFirstChild("HumanoidRootPart")
    return c, hum, hrp
end

local function getVelocity(hrp)
    if not hrp then return 0 end
    local ok, vel = pcall(function() return hrp.AssemblyLinearVelocity end)
    if not ok or not vel then return 0 end
    local horizontal = Vector3.new(vel.X, 0, vel.Z)
    return horizontal.Magnitude
end

local function updateSpeedometerDisplay(velocity)
    if not speedometerValueLabel then return end
    
    local speed = round(velocity, 1)
    currentPlayerSpeed = speed
    speedometerValueLabel.Text = string.format("%.1f", speed)
    
    if currentSpeedLabel then
        currentSpeedLabel.Text = "Current Speed: " .. string.format("%.1f", speed) .. " stud/s"
    end
    
    if speed < 16 then
        speedometerValueLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
        if speedometerStatusLabel then 
            speedometerStatusLabel.Text = "NORMAL SPEED"
            speedometerStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
        end
    elseif speed < 50 then
        speedometerValueLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
        if speedometerStatusLabel then 
            speedometerStatusLabel.Text = "FAST"
            speedometerStatusLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
        end
    elseif speed < 100 then
        speedometerValueLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        if speedometerStatusLabel then 
            speedometerStatusLabel.Text = "VERY FAST"
            speedometerStatusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        end
    else
        speedometerValueLabel.TextColor3 = Color3.fromRGB(255, 50, 50)
        if speedometerStatusLabel then 
            speedometerStatusLabel.Text = "⚡ EXTREME SPEED ⚡"
            speedometerStatusLabel.TextColor3 = Color3.fromRGB(255, 50, 50)
        end
    end
    
    if speed > 100 then
        speedometerValueLabel.Text = "⚡ " .. string.format("%.1f", speed) .. " ⚡"
    end
end

local function toggleSpeedometer()
    if speedometerActive then
        if speedometerConnection then
            speedometerConnection:Disconnect()
            speedometerConnection = nil
        end
        speedometerActive = false
        if speedometerOverlay then
            speedometerOverlay.Visible = false
        end
        showToast("📊 Speedometer: OFF")
    else
        if not speedometerOverlay then
            createSpeedometerOverlay()
        end
        speedometerOverlay.Visible = true
        speedometerActive = true
        
        speedometerConnection = RunService.RenderStepped:Connect(function()
            local c, hum, hrp = getChar()
            if hum and hrp then
                local vel = getVelocity(hrp)
                updateSpeedometerDisplay(vel)
            end
        end)
        showToast("📊 Speedometer: ON")
    end
end

local function getHumanoidState(hum)
    local state = hum:GetState()
    return {
        state = state,
        climbing = (state == Enum.HumanoidStateType.Climbing),
        jumping = (state == Enum.HumanoidStateType.Jumping or state == Enum.HumanoidStateType.Freefall),
        sitting = hum.Sit,
        swimming = (state == Enum.HumanoidStateType.Swimming),
    }
end

-- RECORDING LOOP
RunService.Heartbeat:Connect(function(dt)
    if not isRecording then return end
    sampleAcc = sampleAcc + dt
    if sampleAcc >= SAMPLE_INTERVAL then
        sampleAcc = sampleAcc - SAMPLE_INTERVAL
        local c, hum, hrp = getChar()
        if hum and hrp then
            local t = tick() - recordStartTick
            local humState = getHumanoidState(hum)
            local velocity = getVelocity(hrp)
            local currentWalkSpeed = hum.WalkSpeed

            table.insert(recordedFrames, {
                t = t,
                pos = hrp.Position,
                cframe = hrp.CFrame,
                velocity = velocity,
                walkSpeed = currentWalkSpeed,
                state = humState.state,
                climbing = humState.climbing,
                jumping = humState.jumping,
                sitting = humState.sitting,
                swimming = humState.swimming,
                jump = hum.Jump,
                sit = hum.Sit,
            })
            timerLabel.Text = timeFmt(t)
            totalPlaybackDuration = t
        end
    end
end)

-- ========== SAVE FUNCTION ==========
local function saveRecordingToClipboard()
    if #recordedFrames == 0 then
        showToast("No recording to save")
        return false
    end
    
    local originalSpeedValue = recordedFrames[1] and recordedFrames[1].walkSpeed or DEFAULT_PLAYBACK_SPEED
    
    local simpleData = { 
        version = 3,
        name = "ONIUM Replay Recording",
        createdAt = os.date("%Y-%m-%d %H:%M:%S"),
        frameCount = #recordedFrames,
        duration = totalPlaybackDuration,
        originalWalkSpeed = originalSpeedValue,
        playbackSpeed = currentPlaybackWalkSpeed,
        frames = {} 
    }
    
    for i, frame in ipairs(recordedFrames) do
        local x, y, z, r00, r01, r02, r10, r11, r12, r20, r21, r22 = frame.cframe:GetComponents()
        
        simpleData.frames[i] = {
            t = frame.t, 
            x = x, y = y, z = z,
            r00 = r00, r01 = r01, r02 = r02,
            r10 = r10, r11 = r11, r12 = r12,
            r20 = r20, r21 = r21, r22 = r22,
            v = frame.velocity,
            ws = frame.walkSpeed,
            climbing = frame.climbing, 
            jumping = frame.jumping,
            sitting = frame.sitting, 
            swimming = frame.swimming
        }
    end
    
    local success, jsonData = pcall(function() return HttpService:JSONEncode(simpleData) end)
    
    if success and jsonData then
        if safeClipboard(jsonData) then
            showToast("✅ Saved " .. #recordedFrames .. " frames to clipboard!")
            return true
        else
            showToast("❌ Failed to copy to clipboard")
            return false
        end
    else
        showToast("❌ Failed to encode recording data")
        return false
    end
end

-- ========== LOAD FUNCTION DENGAN GITHUB/PASTEBIN SUPPORT ==========
local function processJSONData(jsonData, source)
    local success, data = pcall(function() return HttpService:JSONDecode(jsonData) end)
    
    if not success or not data or not data.frames then
        showToast("❌ Invalid JSON format")
        return false
    end
    
    if #data.frames < 2 then
        showToast("❌ Recording too short (need at least 2 frames)")
        return false
    end
    
    if playbackActive then
        stopPlayback()
        task.wait(0.1)
    end
    
    if data.originalWalkSpeed then
        originalRecordingWalkSpeed = data.originalWalkSpeed
    else
        originalRecordingWalkSpeed = DEFAULT_PLAYBACK_SPEED
    end
    updateSpeedInfo()
    
    local newFrames = {}
    for i, frameData in ipairs(data.frames) do
        local cframe
        if frameData.r00 then
            cframe = CFrame.new(
                frameData.x or 0, frameData.y or 0, frameData.z or 0,
                frameData.r00 or 1, frameData.r01 or 0, frameData.r02 or 0,
                frameData.r10 or 0, frameData.r11 or 1, frameData.r12 or 0,
                frameData.r20 or 0, frameData.r21 or 0, frameData.r22 or 1
            )
        else
            cframe = CFrame.new(frameData.x or 0, frameData.y or 0, frameData.z or 0)
        end
        
        newFrames[i] = {
            t = frameData.t or (i * SAMPLE_INTERVAL),
            pos = Vector3.new(frameData.x or 0, frameData.y or 0, frameData.z or 0),
            cframe = cframe,
            velocity = frameData.v or 0,
            walkSpeed = frameData.ws or originalRecordingWalkSpeed,
            state = Enum.HumanoidStateType.Running,
            climbing = frameData.climbing or false,
            jumping = frameData.jumping or false,
            sitting = frameData.sitting or false,
            swimming = frameData.swimming or false,
            jump = false,
            sit = frameData.sitting or false
        }
    end
    
    recordedFrames = newFrames
    totalPlaybackDuration = recordedFrames[#recordedFrames].t
    currentPlaybackTime = 0
    lastPlaybackFrameIndex = 1
    playbackPaused = false
    playbackFinished = false
    
    timerLabel.Text = timeFmt(totalPlaybackDuration)
    
    if pathRecordActive then
        showPathRecord()
    end
    
    showToast("✅ Loaded " .. #recordedFrames .. " frames from " .. source .. "!")
    return true
end

-- ========== SMART PLAYBACK RESUME DENGAN ANIMASI KAKI ==========
local function findNearestFrameToPosition(position)
    if #recordedFrames == 0 then return 1, 0 end
    
    local closestIndex = 1
    local closestDistance = (recordedFrames[1].pos - position).Magnitude
    
    for i = 2, #recordedFrames do
        local distance = (recordedFrames[i].pos - position).Magnitude
        if distance < closestDistance then
            closestDistance = distance
            closestIndex = i
        end
    end
    
    return closestIndex, closestDistance
end

local function findFrameAtTime(time)
    if #recordedFrames == 0 then return 1 end
    
    for i = 1, #recordedFrames - 1 do
        if time >= recordedFrames[i].t and time <= recordedFrames[i+1].t then
            return i
        end
    end
    
    return #recordedFrames
end

local function getSmartPlaybackPosition()
    local c, hum, hrp = getChar()
    if not hrp then return 1, 0, "error" end
    
    local currentPos = hrp.Position
    local closestIndex, distance = findNearestFrameToPosition(currentPos)
    
    if playbackFinished then
        playbackFinished = false
        showToast("🎬 Playback finished, restarting from beginning")
        return 1, recordedFrames[1].t, "finished"
    end
    
    if distance > TELEPORT_THRESHOLD then
        showToast("📍 Far from path, teleporting to start")
        return 1, recordedFrames[1].t, "far"
    end
    
    if lastKnownPosition and (lastKnownPosition - currentPos).Magnitude < 15 then
        local timeIndex = findFrameAtTime(currentPlaybackTime)
        showToast("▶️ Resuming from " .. timeFmtDetailed(currentPlaybackTime))
        return timeIndex, currentPlaybackTime, "resume"
    end
    
    local targetTime = recordedFrames[closestIndex].t
    showToast("📍 Found nearest point at " .. timeFmtDetailed(targetTime))
    return closestIndex, targetTime, "nearest"
end

local function stopPlayback()
    if playbackActive then
        playbackActive = false
        if playbackConnection then
            playbackConnection:Disconnect()
            playbackConnection = nil
        end
        
        local c, hum, hrp = getChar()
        if hrp then
            lastKnownPosition = hrp.Position
        end
        
        -- Reset velocity agar karakter berhenti
        if hrp then
            hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        end
        
        if #recordedFrames > 0 and currentPlaybackTime > 0 then
            local nearestIndex = findFrameAtTime(currentPlaybackTime)
            if nearestIndex <= #recordedFrames then
                lastPlaybackFrameIndex = nearestIndex
                currentPlaybackTime = recordedFrames[nearestIndex].t
            end
        end
        
        local c, hum, hrp = getChar()
        if hum then
            hum.WalkSpeed = originalRecordingWalkSpeed or DEFAULT_PLAYBACK_SPEED
            hum.Sit = false
            hum.PlatformStand = false
            -- Reset state humanoid
            hum:ChangeState(Enum.HumanoidStateType.Standing)
        end
        statusText.Text = "⚡ Ready"
        timerLabel.Text = timeFmt(currentPlaybackTime)
        if playbackTimeLabel then
            playbackTimeLabel.Text = "⏸️ " .. timeFmtDetailed(currentPlaybackTime) .. " / " .. timeFmtDetailed(totalPlaybackDuration)
        end
        playbackPaused = true
    end
end

-- ========== FUNGSI PLAYBACK UTAMA DENGAN ANIMASI KAKI ==========
local function startPlayback()
    if #recordedFrames < 2 then
        showToast("No recording")
        return false
    end

    if playbackActive then
        stopPlayback()
        task.wait(0.1)
    end

    local c, hum, hrp = getChar()
    if not hum or not hrp then
        showToast("Character not ready")
        return false
    end

    local startIndex, startTime, reason = getSmartPlaybackPosition()
    
    currentPlaybackTime = startTime
    lastPlaybackFrameIndex = startIndex
    playbackPaused = false
    
    local targetFrame = recordedFrames[startIndex]
    if targetFrame and targetFrame.cframe then
        hrp.CFrame = targetFrame.cframe
    end

    originalRecordingWalkSpeed = hum.WalkSpeed
    updateSpeedInfo()

    hum.Sit = false
    hum.PlatformStand = false
    -- Aktifkan AutoRotate agar karakter bisa berjalan normal
    hum.AutoRotate = true
    task.wait(0.05)

    playbackActive = true
    local recordedBaseSpeed = recordedFrames[1] and recordedFrames[1].walkSpeed or DEFAULT_PLAYBACK_SPEED
    
    statusText.Text = "▶️ Playing... (Kaki bergerak!)"

    playbackConnection = RunService.Heartbeat:Connect(function(deltaTime)
        if not playbackActive then 
            if playbackConnection then playbackConnection:Disconnect() end
            playbackConnection = nil
            return 
        end
        
        local speedMultiplier = currentPlaybackWalkSpeed / recordedBaseSpeed
        if speedMultiplier < 0.1 then speedMultiplier = 0.1 end
        if speedMultiplier > 10 then speedMultiplier = 10 end
        
        currentPlaybackTime = currentPlaybackTime + (deltaTime * speedMultiplier)
        
        if currentPlaybackTime >= totalPlaybackDuration then
            if isLoopMode then
                currentPlaybackTime = currentPlaybackTime - totalPlaybackDuration
                lastPlaybackFrameIndex = 1
                if recordedFrames[1] and recordedFrames[1].cframe then
                    hrp.CFrame = recordedFrames[1].cframe
                end
                showToast("🔄 Loop restarting")
            else
                playbackFinished = true
                stopPlayback()
                showToast("🏁 Playback finished - Press PLAY to start from beginning")
                return
            end
        end
        
        local currentIndex = 1
        for i = 1, #recordedFrames - 1 do
            if currentPlaybackTime >= recordedFrames[i].t and currentPlaybackTime <= recordedFrames[i+1].t then
                currentIndex = i
                break
            end
        end
        
        lastPlaybackFrameIndex = currentIndex
        
        local frameA = recordedFrames[currentIndex]
        local frameB = recordedFrames[currentIndex + 1]
        
        if frameA and frameB then
            local alpha = (currentPlaybackTime - frameA.t) / (frameB.t - frameA.t)
            local newCFrame = frameA.cframe:Lerp(frameB.cframe, alpha)
            
            -- Hitung jarak dan arah pergerakan
            local prevPos = frameA.pos
            local nextPos = frameB.pos
            local distanceBetween = (nextPos - prevPos).Magnitude
            local timeBetween = frameB.t - frameA.t
            
            -- INI KUNCI AGAR KAKI BERGERAK!
            if timeBetween > 0 and distanceBetween > 0.05 then
                local requiredSpeed = distanceBetween / timeBetween
                local direction = (nextPos - prevPos).unit
                
                -- Set walkspeed humanoid
                hum.WalkSpeed = currentPlaybackWalkSpeed
                
                -- BERIKAN VELOCITY AGAR ANIMASI KAKI BERGERAK
                -- Ini adalah bagian yang paling penting!
                local targetVelocity = direction * requiredSpeed * speedMultiplier
                
                -- Terapkan velocity ke root part
                if targetVelocity.Magnitude > 0 then
                    hrp.AssemblyLinearVelocity = targetVelocity
                end
                
                -- Set state humanoid agar animasi berjalan
                if requiredSpeed > 3 then
                    hum:ChangeState(Enum.HumanoidStateType.Running)
                elseif requiredSpeed > 0.5 then
                    hum:ChangeState(Enum.HumanoidStateType.Running)
                else
                    hum:ChangeState(Enum.HumanoidStateType.Standing)
                end
            end
            
            -- Set posisi CFrame (tetap diperlukan untuk akurasi)
            hrp.CFrame = newCFrame
            
            -- Atur state humanoid lainnya (lompat, duduk, dll)
            if frameB.sitting then
                hum.Sit = true
                hum.Jump = false
                hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
            elseif frameB.jumping then
                hum.Sit = false
                hum:ChangeState(Enum.HumanoidStateType.Jumping)
                hum.Jump = true
            elseif frameB.climbing then
                hum.Sit = false
                hum:ChangeState(Enum.HumanoidStateType.Climbing)
            elseif frameB.swimming then
                hum.Sit = false
                hum:ChangeState(Enum.HumanoidStateType.Swimming)
            end
        end
        
        timerLabel.Text = timeFmt(currentPlaybackTime)
        if playbackTimeLabel then
            playbackTimeLabel.Text = timeFmtDetailed(currentPlaybackTime) .. " / " .. timeFmtDetailed(totalPlaybackDuration)
        end
    end)
    
    return true
end

-- Block F key
local function blockFKey(input, gameProcessed)
    if input.KeyCode == Enum.KeyCode.F then
        return true
    end
    return false
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if blockFKey(input, gameProcessed) then
        return
    end
end)

UserInputService.InputChanged:Connect(function(input, gameProcessed)
    if blockFKey(input, gameProcessed) then
        return
    end
end)

if ContextActionService then
    ContextActionService:BindAction("BlockFKey", function(actionName, inputState, inputObject)
        if inputObject.KeyCode == Enum.KeyCode.F then
            return Enum.ContextActionResult.Sink
        end
        return Enum.ContextActionResult.Pass
    end, false, Enum.KeyCode.F)
end
